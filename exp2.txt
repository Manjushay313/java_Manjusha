public class Demo {
   
    static void show() {
        System.out.println("Static method with no parameters");
    }

    static void show(int x) {
        System.out.println("Static method with one int parameter: " + x);
    }

    static void show(String s, int y) {
        System.out.println("Static method with String and int: " + s + ", " + y);
    }

    public static void main(String[] args) {
        Demo.show();
        Demo.show(10);
        Demo.show("Manjusha", 21);
    }
}
